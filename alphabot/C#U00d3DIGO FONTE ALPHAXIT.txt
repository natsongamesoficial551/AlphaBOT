        // 1. HANDLE ROTATIVO COM ACESSO MÍNIMO VARIÁVEL
        private IntPtr GetRotatingProcessHandle(string processName)
        {
            try
            {
                var processes = Process.GetProcessesByName(processName);
                if (processes.Length == 0) return IntPtr.Zero;
                int pid = processes[0].Id;

                // Array de combinações de permissões mínimas
                uint[] permissionSets = new uint[]
                {
                    0x0010, // PROCESS_VM_READ
                    0x0010 | 0x0020, // READ + WRITE
                    0x0010 | 0x0008, // READ + OPERATION
                    0x0010 | 0x0400, // READ + QUERY
                    0x0020 | 0x0008, // WRITE + OPERATION
                    0x0010 | 0x0020 | 0x0008, // READ + WRITE + OPERATION
                    0x0010 | 0x0020 | 0x0400, // READ + WRITE + QUERY
                };

                // Escolhe aleatoriamente um conjunto de permissões
                uint desiredAccess = permissionSets[antiBanRandom.Next(permissionSets.Length)];

                IntPtr hProcess = OpenProcess(desiredAccess, false, pid);

                // Adiciona delay aleatório antes de retornar
                Thread.Sleep(antiBanRandom.Next(1, 10));
                return hProcess;
            }
            catch
            {
                return IntPtr.Zero;
            }
        }

        // 3. ANTI-ANÁLISE AVANÇADA (TIMING + AMBIENTE)
        private bool IsAnalysisEnvironment()
        {
            try
            {
                // Verificação de timing (sandbox detection)
                Stopwatch sw = Stopwatch.StartNew();
                double dummy = 0;
                for (int i = 0; i < 1000000; i++)
                {
                    dummy += Math.Sin(i) * Math.Cos(i);
                }
                sw.Stop();

                // Se execução muito rápida (menos de 10ms), possível sandbox
                if (sw.ElapsedMilliseconds < 10)
                    return true;

                // Verifica quantidade de núcleos (VMs geralmente têm menos)
                int coreCount = Environment.ProcessorCount;
                if (coreCount < 2)
                    return true;

                // Verifica resolução da tela (sandboxes podem ter resolução baixa)
                Screen primaryScreen = Screen.PrimaryScreen;
                if (primaryScreen.Bounds.Width < 1024 || primaryScreen.Bounds.Height < 768)
                    return true;

                // Verifica processos de análise
                string[] analysisProcesses = {
                    "vmsrvc", "vmusrvc", "vmwaretray", "vmwareuser",
                    "vboxservice", "vboxtray", "xenservice", "qemu-ga",
                    "prl_cc", "prl_tools", "vmtoolsd", "df5serv"
                };

                foreach (string proc in analysisProcesses)
                {
                    if (Process.GetProcessesByName(proc).Length > 0)
                        return true;
                }

                // Verifica debuggers existentes
                if (IsDebuggerPresent())
                    return true;

                return false;
            }
            catch
            {
                return true; // Em caso de erro, assume ambiente suspeito
            }
        }

        // 4. OFUSCAÇÃO DE STRINGS COM DECRYPT EM RUNTIME
        private string DecryptString(byte[] encryptedBytes, byte key)
        {
            byte[] decrypted = new byte[encryptedBytes.Length];
            for (int i = 0; i < encryptedBytes.Length; i++)
            {
                decrypted[i] = (byte)(encryptedBytes[i] ^ key);
            }
            return Encoding.UTF8.GetString(decrypted);
        }

        // 6. LIMPEZA DE RASTROS COM MÚLTIPLOS PASSES ALEATÓRIOS APERFEIÇOADO
        private void EnhancedCleanMemoryTraces(string cheatName)
        {
            lock (memoryLock) // THREAD SAFE - Evita conflitos
            {
                if (!originalValues.ContainsKey(cheatName)) return;

                var backupDict = originalValues[cheatName];
                if (backupDict == null || backupDict.Count == 0) return;

                // Verifica se o processo ainda existe
                if (Process.GetProcessesByName("HD-Player").Length == 0)
                {
                    originalValues.Remove(cheatName);
                    return;
                }

                try
                {
                    memory.OpenProcess("HD-Player");

                    foreach (var kvp in backupDict)
                    {
                        try
                        {
                            long address = kvp.Key;
                            byte[] originalBytes = kvp.Value;

                            if (originalBytes == null || originalBytes.Length == 0) continue;

                            string originalString = BitConverter.ToString(originalBytes).Replace("-", " ");

                            // TIMEOUT para não travar (100ms)
                            var task = Task.Run(() => memory.WriteMemory(address.ToString("X"), "bytes", originalString));
                            if (!task.Wait(100)) continue; // Pula se travar

                            Thread.Sleep(antiBanRandom.Next(5, 15));
                        }
                        catch { /* Ignora erro individual, continua tentando outros */ }
                    }
                }
                catch { /* Falha silenciosa */ }
                finally
                {
                    originalValues.Remove(cheatName);
                }
            }
        }

        // Limpeza de diretórios
        private int CleanDirectorySafe(string path)
        {
            int removed = 0;
            try
            {
                if (!Directory.Exists(path))
                    return 0;
                foreach (string file in Directory.GetFiles(path))
                {
                    try
                    {
                        File.SetAttributes(file, FileAttributes.Normal);
                        File.Delete(file);
                        removed++;
                    }
                    catch { }
                }
                foreach (string dir in Directory.GetDirectories(path))
                {
                    try
                    {
                        Directory.Delete(dir, true);
                        removed++;
                    }
                    catch { }
                }
            }
            catch { }
            return removed;
        }

        // Métodos da tela de login
        private void RealizarLogin()
        {
            if (string.IsNullOrWhiteSpace(Username.Text) || string.IsNullOrWhiteSpace(Pass.Text))
            {
                statusLogin.Text = "Preencha usuário e senha!";
                return;
            }

            // Pede a Auth Key numa janela separada
            string authKey = "";
            using (Form popup = new Form())
            {
                popup.Text = "Alpha Xit — Auth Key";
                popup.Size = new System.Drawing.Size(420, 160);
                popup.StartPosition = FormStartPosition.CenterScreen;
                popup.FormBorderStyle = FormBorderStyle.FixedDialog;
                popup.MaximizeBox = false;
                popup.MinimizeBox = false;

                Label lbl = new Label()
                {
                    Text = "Digite sua Auth Key (recebida no Discord):",
                    Left = 10,
                    Top = 15,
                    Width = 390
                };

                TextBox txt = new TextBox()
                {
                    Left = 10,
                    Top = 40,
                    Width = 380
                };

                Button btn = new Button()
                {
                    Text = "Confirmar",
                    Left = 150,
                    Top = 75,
                    Width = 100
                };

                btn.Click += (s, ev) => { popup.DialogResult = DialogResult.OK; };

                popup.Controls.AddRange(new Control[] { lbl, txt, btn });
                popup.AcceptButton = btn;

                if (popup.ShowDialog() == DialogResult.OK)
                    authKey = txt.Text.Trim();
            }

            if (string.IsNullOrWhiteSpace(authKey))
            {
                statusLogin.Text = "Auth Key não informada!";
                return;
            }

            // REALIZA O LOGIN
            statusLogin.Text = "Autenticando...";

            MyAuthApp.activate(Username.Text.Trim(), Pass.Text.Trim(), authKey);

            if (MyAuthApp.response.success)
            {
                statusLogin.Text = "Auth Key ativada com sucesso!";

                // Esconde o painel de login
                l1.Visible = false;
                l1.SendToBack();

                // Mostra o painel principal
                p1.BringToFront();
                p1.Visible = true;

                status.Text = $"Bem-vindo {MyAuthApp.user_data.username}! " +
                              $"Tempo restante: {MyAuthApp.user_data.tempo_restante}";

                // Inicia o heartbeat
                MyAuthApp.StartHeartbeat();

                // Log de ativação
                MyAuthApp.log($"Usuário {MyAuthApp.user_data.username} ativou o painel");

                Console.Beep();
            }
            else
            {
                statusLogin.Text = MyAuthApp.response.message;
            }
        }

                        // offset 0xA8: endereço de escrita que funciona para ambos

                        Int64 repParado = current + 0xA0;   // Valor para inimigo parado
                        Int64 repMovimento = current + 0xA4; // Valor para inimigo em movimento
                        Int64 repEscrita = current + 0xA8;    // Endereço de escrita

                            try
                            {
                                // Se falhar, tenta do offset de parado
                                Readmem = memory.ReadMemory<int>(repParado.ToString("X"));

                        WriteAimbotMemory(repEscrita.ToString("X"), Readmem.ToString(), "int");
                        anySuccess = true;

                    // Verifica se o scan principal não encontrou nada, tenta scan alternativo
                    if (!anySuccess)

                        foreach (var current in ScanAlternativo)
                        {
                            Int64 repEscritaAlt = current + 0x7C;
                            Int64 repLeituraAlt = current + 0x7A;

                            var ReadmemAlt = memory.ReadMemory<int>(repLeituraAlt.ToString("X"));

                            if (!originalValues.ContainsKey("Rage") || !originalValues["Rage"].ContainsKey(repEscritaAlt))
                            {
                                byte[] originalBytes = memory.ReadBytes(repEscritaAlt.ToString("X"), 4);

                }

            else

        // ========== OTIMIZAÇÃO SUPER AVANÇADA ==========
        // Classe de otimização profissional
        public class AdvancedOptimizer
        {
            [DllImport("kernel32.dll")]
            private static extern bool SetProcessWorkingSetSize(IntPtr proc, int min, int max);

            [DllImport("kernel32.dll")]
            private static extern IntPtr GetCurrentProcess();

            [DllImport("kernel32.dll")]
            private static extern bool SetPriorityClass(IntPtr hProcess, uint dwPriorityClass);

            [DllImport("kernel32.dll")]
            private static extern bool SetProcessAffinityMask(IntPtr hProcess, IntPtr dwProcessAffinityMask);

            [DllImport("ntdll.dll")]
            private static extern int NtSetInformationProcess(IntPtr hProcess, int infoClass, ref int info, int infoSize);

            [DllImport("kernel32.dll")]
            private static extern IntPtr OpenProcess(uint dwDesiredAccess, bool bInheritHandle, int dwProcessId);

            [DllImport("kernel32.dll", SetLastError = true)]
            private static extern bool SetProcessWorkingSetSizeEx(IntPtr hProcess, int dwMinimumWorkingSetSize, int dwMaximumWorkingSetSize, uint Flags);

            [DllImport("user32.dll", SetLastError = true)]
            private static extern bool SystemParametersInfo(int uiAction, int uiParam, int pvParam, int fWinIni);

            [DllImport("kernel32.dll", SetLastError = true)]
            private static extern bool CloseHandle(IntPtr hObject);

            [DllImport("psapi.dll", SetLastError = true)]
            private static extern bool EmptyWorkingSet(IntPtr hProcess);

            [DllImport("winmm.dll")]
            private static extern int timeBeginPeriod(int period);

            [DllImport("winmm.dll")]
            private static extern int timeEndPeriod(int period);

            private const uint PROCESS_SET_INFORMATION = 0x0200;
            private const uint PROCESS_QUERY_INFORMATION = 0x0400;
            private const int MEMORY_PRIORITY_HIGH = 1;

            private Process emulatorProcess;
            private System.Windows.Forms.Timer optimizationTimer;
            private bool isOptimized = false;
            private Random optRandom = new Random();

            public AdvancedOptimizer()
            {
                optimizationTimer = new System.Windows.Forms.Timer();
                optimizationTimer.Interval = 5000;
                optimizationTimer.Tick += OptimizeTick;
            }

            public bool ApplyOptimization(bool enable)
            {
                try
                {
                    var processes = Process.GetProcessesByName("HD-Player");
                    if (processes.Length == 0) return false;

                    emulatorProcess = processes[0];

                    if (enable)
                    {
                        SafeExecute(OptimizeMemoryUsage);
                        SafeExecute(OptimizeCPUAffinity);
                        SafeExecute(OptimizeThreadPriority);
                        SafeExecute(OptimizeGPUUsage);
                        SafeExecute(OptimizeSystemCache);
                        SafeExecute(DisableUnnecessaryFeatures);
                        SafeExecute(ApplyGameMode);
                        SafeExecute(OptimizePowerSettings);

                        optimizationTimer.Start();
                        isOptimized = true;
                        return true;
                    }
                    else
                    {
                        optimizationTimer.Stop();
                        SafeExecute(RestoreDefaults);
                        isOptimized = false;
                        return true;
                    }
                }
                catch
                {
                    return false;
                }
            }

            private void SafeExecute(Action action)
            {
                try
                {
                    action();
                }
                catch (Exception ex)
                {
                    System.Diagnostics.Debug.WriteLine($"Otimização falhou: {ex.Message}");
                }
            }

            private void OptimizeMemoryUsage()
            {
                try
                {
                    SetProcessWorkingSetSize(emulatorProcess.Handle, -1, -1);

                    IntPtr hProcess = OpenProcess(PROCESS_SET_INFORMATION | PROCESS_QUERY_INFORMATION, false, emulatorProcess.Id);
                    if (hProcess != IntPtr.Zero)
                    {
                        int memPriority = MEMORY_PRIORITY_HIGH;
                        NtSetInformationProcess(hProcess, 0x1F, ref memPriority, sizeof(int));

                        int minWorkingSet = 500 * 1024 * 1024;
                        int maxWorkingSet = int.MaxValue;

                        SetProcessWorkingSetSizeEx(hProcess, minWorkingSet, maxWorkingSet, 0);
                        CloseHandle(hProcess);
                    }

                    EmptyWorkingSet(GetCurrentProcess());

                    try
                    {
                        Registry.SetValue(@"HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\Session Manager\Memory Management", "LargeSystemCache", 1);
                        Registry.SetValue(@"HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\Session Manager\Memory Management", "DisablePagingExecutive", 1);
                    }
                    catch { }
                }
                catch { }
            }

            private void OptimizeCPUAffinity()
            {
                try
                {
                    int coreCount = Environment.ProcessorCount;

                    if (coreCount >= 4)
                    {
                        long affinityMask = 0;
                        for (int i = 2; i < coreCount; i++)
                        {
                            affinityMask |= (1L << i);
                        }

                        if (affinityMask != 0)
                        {
                            SetProcessAffinityMask(emulatorProcess.Handle, (IntPtr)affinityMask);
                            SetPriorityClass(emulatorProcess.Handle, 0x00000080);
                        }
                    }
                }
                catch { }
            }

            private void OptimizeThreadPriority()
            {
                try
                {
                    Process.GetCurrentProcess().PriorityClass = ProcessPriorityClass.High;
                    timeBeginPeriod(1);

                    try
                    {
                        Registry.SetValue(@"HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows\Windows Error Reporting", "Disabled", 1);
                        Registry.SetValue(@"HKEY_CURRENT_USER\Software\Microsoft\Windows\Windows Error Reporting", "Disabled", 1);
                    }
                    catch { }
                }
                catch { }
            }

            private void OptimizeGPUUsage()
            {
                try
                {
                    try
                    {
                        Registry.SetValue(@"HKEY_CURRENT_USER\Software\Microsoft\DirectX\GraphicsSettings", "PreferExternalManifest", 1, RegistryValueKind.DWord);
                        Registry.SetValue(@"HKEY_CURRENT_USER\Software\Microsoft\Direct3D", "EnableDX9Optimizations", 1);
                        Registry.SetValue(@"HKEY_CURRENT_USER\Software\Microsoft\DirectX", "DisableVSync", 1);
                        Registry.SetValue(@"HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\DirectX", "MaxFrameRate", "0");
                    }
                    catch { }
                }
                catch { }
            }

            private void OptimizeSystemCache()
            {
                try
                {
                    try
                    {
                        Registry.SetValue(@"HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\FileSystem", "NtfsDisableLastAccessUpdate", 1);
                        Registry.SetValue(@"HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\FileSystem", "NtfsDisable8dot3NameCreation", 1);
                    }
                    catch { }
                }
                catch { }
            }

            private void DisableUnnecessaryFeatures()
            {
                try
                {
                    int disableAnimation = 0;
                    int disableEffect = 0;

                    SystemParametersInfo(0x0042, 0, disableAnimation, 0);
                    SystemParametersInfo(0x0043, 0, disableAnimation, 0);
                    SystemParametersInfo(0x0044, 0, disableAnimation, 0);
                    SystemParametersInfo(0x0022, 0, disableEffect, 0);
                }
                catch { }
            }

            private void ApplyGameMode()
            {
                try
                {
                    try
                    {
                        Registry.SetValue(@"HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\GameDVR", "AppCaptureEnabled", 0);
                        Registry.SetValue(@"HKEY_CURRENT_USER\System\GameConfigStore", "GameDVR_Enabled", 0);
                    }
                    catch { }
                }
                catch { }
            }

            private void OptimizePowerSettings()
            {
                try
                {
                    try
                    {
                        Registry.SetValue(@"HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\Power\User\PowerSchemes", "ActivePowerScheme", "8c5e7fda-e8bf-4a96-9a85-a6e23a8c635c");
                    }
                    catch { }
                }
                catch { }
            }

            private void RestoreDefaults()
            {
                try
                {
                    if (emulatorProcess != null && !emulatorProcess.HasExited)
                    {
                        SetPriorityClass(emulatorProcess.Handle, 0x00000020);
                        SetProcessWorkingSetSize(emulatorProcess.Handle, -1, -1);
                    }

                    try
                    {
                        timeEndPeriod(1);
                    }
                    catch { }
                }
                catch { }
            }

            private void OptimizeTick(object sender, EventArgs e)
            {
                if (isOptimized && emulatorProcess != null && !emulatorProcess.HasExited)
                {
                    SafeExecute(OptimizeMemoryUsage);
                    Thread.Sleep(optRandom.Next(100, 300));
                }
            }
        }

        // ========== MOUSEFIX PROFISSIONAL ==========
        public class MouseFix
        {
            [DllImport("user32.dll")]
            private static extern int SystemParametersInfo(int uAction, int uParam, int lpvParam, int fuWinIni);

            [DllImport("user32.dll")]
            private static extern bool SystemParametersInfo(int uAction, int uParam, out int lpvParam, int fuWinIni);

            [DllImport("user32.dll")]
            private static extern bool RegisterRawInputDevices(ref RAWINPUTDEVICE pRawInputDevice, uint uiNumDevices, uint cbSize);

            [DllImport("user32.dll")]
            private static extern bool SetProcessDPIAware();

            private const int SPI_GETMOUSE = 0x0003;
            private const int SPI_SETMOUSE = 0x0004;
            private const int SPI_GETMOUSESPEED = 0x0070;
            private const int SPI_SETMOUSESPEED = 0x0071;

            private int[] originalMouseSpeed = new int[3];
            private int originalMouseThreshold;

            public bool ApplyMouseFix(bool enable)
            {
                try
                {
                    if (enable)
                    {
                        SaveOriginalSettings();
                        RemoveMouseAcceleration();
                        DisableEnhancedPointerPrecision();
                        OptimizeRawInput();
                        ConfigureDPISettings();
                        return true;
                    }
                    else
                    {
                        RestoreOriginalSettings();
                        return true;
                    }
                }
                catch
                {
                    return false;
                }
            }

            private void SaveOriginalSettings()
            {
                try
                {
                    SystemParametersInfo(SPI_GETMOUSE, 0, out originalMouseSpeed[0], 0);
                    int speed;
                    SystemParametersInfo(SPI_GETMOUSESPEED, 0, out speed, 0);
                    originalMouseThreshold = speed;
                }
                catch { }
            }

            private void RemoveMouseAcceleration()
            {
                try
                {
                    int newSetting = 0;
                    SystemParametersInfo(SPI_SETMOUSE, 0, newSetting, 0x02);
                    int gameSpeed = 10;
                    SystemParametersInfo(SPI_SETMOUSESPEED, 0, gameSpeed, 0x02);
                }
                catch { }
            }

            private void DisableEnhancedPointerPrecision()
            {
                try
                {
                    Registry.SetValue(@"HKEY_CURRENT_USER\Control Panel\Mouse", "MouseSpeed", "0");
                    Registry.SetValue(@"HKEY_CURRENT_USER\Control Panel\Mouse", "MouseThreshold1", "0");
                    Registry.SetValue(@"HKEY_CURRENT_USER\Control Panel\Mouse", "MouseThreshold2", "0");
                }
                catch { }
            }

            private void OptimizeRawInput()
            {
                try
                {
                    RAWINPUTDEVICE rid = new RAWINPUTDEVICE();
                    rid.usUsagePage = 0x01;
                    rid.usUsage = 0x02;
                    rid.dwFlags = 0x00000100;
                    rid.hwndTarget = IntPtr.Zero;

                    RegisterRawInputDevices(ref rid, 1, (uint)Marshal.SizeOf(typeof(RAWINPUTDEVICE)));
                    Registry.SetValue(@"HKEY_CURRENT_USER\Control Panel\Mouse", "MouseHoverTime", "1");
                }
                catch { }
            }

            private void ConfigureDPISettings()
            {
                try
                {
                    string emulatorPath = @"C:\Program Files\Microvirt\HD-Player.exe";
                    if (File.Exists(emulatorPath))
                    {
                        Registry.SetValue(@"HKEY_CURRENT_USER\Software\Microsoft\Windows NT\CurrentVersion\AppCompatFlags\Layers",
                            emulatorPath, "~ HIGHDPIAWARE");
                    }
                    SetProcessDPIAware();
                }
                catch { }
            }

            private void RestoreOriginalSettings()
            {
                try
                {
                    SystemParametersInfo(SPI_SETMOUSE, 0, originalMouseSpeed[0], 0x02);
                    SystemParametersInfo(SPI_SETMOUSESPEED, 0, originalMouseThreshold, 0x02);

                    Registry.SetValue(@"HKEY_CURRENT_USER\Control Panel\Mouse", "MouseSpeed", "1");
                    Registry.SetValue(@"HKEY_CURRENT_USER\Control Panel\Mouse", "MouseThreshold1", "6");
                    Registry.SetValue(@"HKEY_CURRENT_USER\Control Panel\Mouse", "MouseThreshold2", "10");
                }
                catch { }
            }
        }

        private bool AplicarOtimizacaoSegura()
        {
            try
            {
                var emulator = Process.GetProcessesByName("HD-Player").FirstOrDefault();
                if (emulator != null)
                {
                    // Aumenta prioridade do processo (seguro)
                    try { emulator.PriorityClass = ProcessPriorityClass.High; } catch { }

                    // Define afinidade para usar todos os núcleos
                    try
                    {
                        int coreCount = Environment.ProcessorCount;
                        long mask = (1L << coreCount) - 1;
                        SetProcessAffinityMask(emulator.Handle, (IntPtr)mask);
                    }
                    catch { }
                }

                // Otimizar o processo atual (o painel)
                try { Process.GetCurrentProcess().PriorityClass = ProcessPriorityClass.AboveNormal; } catch { }

                // Aumentar resolução do timer do sistema
                try { timeBeginPeriod(1); } catch { }

                // Limpar working set do emulador (libera memória)
                try { if (emulator != null) EmptyWorkingSet(emulator.Handle); } catch { }

                return true;
            }
            catch { return false; }
        }

        private void RestaurarOtimizacaoSegura()
        {
            try
            {
                var emulator = Process.GetProcessesByName("HD-Player").FirstOrDefault();
                if (emulator != null)
                {
                    try { emulator.PriorityClass = ProcessPriorityClass.Normal; } catch { }
                }

                try { Process.GetCurrentProcess().PriorityClass = ProcessPriorityClass.Normal; } catch { }
                try { timeEndPeriod(1); } catch { }
            }
            catch { }
        }

        private void ToggleBetweenGameAndPanel()
        {
            // Verifica se o jogo do dinossauro está visível/ativo
            bool isGameVisible = (dinoGame != null && !dinoGame.IsDisposed && dinoGame.Visible);
            bool isPanelVisible = (this.Visible && this.WindowState != FormWindowState.Minimized);

            if (isGameVisible)
            {
                // F6 PRESSIONADO - JOGO ESTÁ VISÍVEL -> VAI PARA O PAINEL
                CloseDinoGameAndShowPanel();
            }
            else if (isPanelVisible)
            {
                // F6 PRESSIONADO - PAINEL ESTÁ VISÍVEL -> VAI PARA O JOGO
                HidePanelAndStartDinoGame();
            }
            else
            {
                // Nenhum dos dois está visível (caso de erro) - mostra o jogo
                HidePanelAndStartDinoGame();
            }
        }

        private void StartDinoGame()
        {
            if (dinoGame == null || dinoGame.IsDisposed)

        private void StartF6TimerForPanel()
        {
            if (f6Timer != null)
            {
                f6Timer.Stop();
                f6Timer.Dispose();
            }

            f6Timer = new System.Windows.Forms.Timer();
            f6Timer.Interval = 100;

            DateTime? holdStart = null;
            bool trocando = false;  // Impede trocas enquanto já está trocando

            f6Timer.Tick += (sender, e) =>
            {
                if (trocando) return;  // Se já está trocando, ignora

                bool f6Pressed = (GetAsyncKeyState(0x75) & 0x8000) != 0;

                if (f6Pressed && holdStart == null)
                {
                    holdStart = DateTime.Now;
                }
                else if (f6Pressed && holdStart != null)
                {
                    double elapsed = (DateTime.Now - holdStart.Value).TotalSeconds;
                    if (elapsed >= 10)
                    {
                        trocando = true;
                        holdStart = null;
                        HidePanelAndStartDinoGame();
                        // Libera depois de 2 segundos
                        System.Windows.Forms.Timer releaseTimer = new System.Windows.Forms.Timer();
                        releaseTimer.Interval = 2000;
                        releaseTimer.Tick += (s, ev) =>
                        {
                            trocando = false;
                            releaseTimer.Stop();
                            releaseTimer.Dispose();
                        };
                        releaseTimer.Start();
                    }
                }
                else if (!f6Pressed)
                {
                    holdStart = null;
                }
            };
            f6Timer.Start();
        }

        private void CloseDinoGameAndShowPanel()
        {
            // Fecha o jogo primeiro
            if (dinoGame != null && !dinoGame.IsDisposed)
            {
                dinoGame.Close();
                dinoGame = null;
            }

            // Pequeno delay para garantir que o jogo fechou
            Thread.Sleep(100);

            // Mostra o painel
            this.ShowInTaskbar = true;
            this.WindowState = FormWindowState.Normal;
            this.Show();
            this.BringToFront();
            this.Focus();

            // Força atualização visual
            Application.DoEvents();

            // Restaura timers do cheat
            if (aimAssistTimer != null && aimAssistActive)
                aimAssistTimer.Start();
            if (securityTimer != null)
                securityTimer.Start();

            status.Text = "Painel aberto. Segure F6 por 10 segundos para voltar ao jogo.";
        }

        private void HidePanelAndStartDinoGame()
        {
            // Para timers do cheat
            if (aimAssistTimer != null)
                aimAssistTimer.Stop();
            if (securityTimer != null)
                securityTimer.Stop();

            // Esconde o painel
            this.ShowInTaskbar = false;

            if (dinoGame != null && !dinoGame.IsDisposed)

                dinoGame.Close();
                dinoGame = null;
            }

            // Pequeno delay
            Thread.Sleep(100);

            // Abre o jogo do dinossauro
            StartDinoGame();
        }

            private void GameTimer_Tick(object sender, EventArgs e)
            {
                if (!gameRunning || isGameOver) return;

                // Movimento do cacto
                cactus.Left -= cactusSpeed;

                // Gerar novo cacto
                if (cactus.Right < 0)
                {
                    cactus.Left = this.ClientSize.Width;
                    score++;
                    scoreLabel.Text = "Score: " + score;

                    // Aumenta dificuldade gradualmente
                    if (score % 10 == 0 && cactusSpeed < 20)
                    {
                        cactusSpeed += 1;
                    }
                }

                // Colisão
                if (dino.Bounds.IntersectsWith(cactus.Bounds))
                {
                    GameOver();
                }

                // Gravidade
                if (!isJumping && dino.Top < dinoGroundY)
                {
                    dino.Top += gravity;
                    if (dino.Top >= dinoGroundY)
                    {
                        dino.Top = dinoGroundY;
                    }
                }
            }

            private void JumpTimer_Tick(object sender, EventArgs e)
            {
                if (!isJumping || isGameOver) return;

                dino.Top -= jumpSpeed;

                if (dino.Top <= dinoGroundY - 80)
                {
                    jumpTimer.Stop();
                }
            }
